# Simplified TSP Algorithm
def find_route():
    route = ['WH']
    while len(route) < 7:  # 6 towns + WH
        current = route[-1]
        # Find nearest node not in route
        next_node = min(
            [n for n in nodes if n not in route],
            key=lambda x: distances[current][x]
        )
        route.append(next_node)
    return route + ['WH']
