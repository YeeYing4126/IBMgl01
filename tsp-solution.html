Optimization Algorithm
def calculate_optimal_route():
    # Valid connections with distances
    distances = {
        ('WH', 'A'): 10, ('A', 'WH'): 10,
        ('A', 'B'): 8,  ('B', 'A'): 8,
        ('B', 'C'): 7,  ('C', 'B'): 7,
        ('C', 'D'): 6,  ('D', 'C'): 6,
        ('D', 'F'): 5,  ('F', 'D'): 5,
        ('F', 'E'): 8,  ('E', 'F'): 8,
        ('E', 'WH'): 22,('WH', 'E'): 22
    }

    route = ['WH', 'A', 'B', 'C', 'D', 'F', 'E', 'WH']
    total = 0
    for i in range(len(route)-1):
        total += distances[(route[i], route[i+1])]
    
    return total, route

total_distance, optimal_route = calculate_optimal_route()
print(f"Optimal Route: {' → '.join(optimal_route)}")
print(f"Total Distance: {total_distance}km")
